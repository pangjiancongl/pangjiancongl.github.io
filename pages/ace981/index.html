    最长有效括号

    # 需求

    给你一个只包含 '(' 和 ')' 的字符串，找出最长有效（格式正确且连续）括号子串的长度。

    # 示例

    1.
输入：s = "(()"
输出：2
解释：最长有效括号子串是 "()"
2.
输入：s = ")()())"
输出：4
解释：最长有效括号子串是 "()()"
3.
输入：s = ""
输出：0

    # 代码

    /**
 * @param {string} s
 * @return {number}
 */
var longestValidParentheses = function (s) {
  // 使用到的方法
  // push 末尾添加
  // pop 末尾删除
  // 栈：只能从一端进或者出
  // 存放最大值
  let max = 0;
  // 数组长度
  let len = s.length;
  // 存放'()'
  let arr = [];
  // 默认栈存入-1占位
  // 为什么用-1占位，因为数组的索引是从0开始的,当字符串为'()'时1-0的结果为1 但实际是有两个长度的，这里用-1可以实现+1的效果
  // 为什么需要占位 当第一个字符为）时会执行arr.pop() 如果没有占位会报错
  arr.push(-1);
  for (let i = 0; i < len; i++) {
    // 字符串里面的每个元素
    let item = s[i];
    // 判断是不是（
    if (item === "(") {
      // 是的话就存入到预先准备的栈里面
      arr.push(i);
    } else {
      // 如果是)就把存进去的(移除
      arr.pop();
      // 存储arr的长度减少下面的获取次数
      let c = arr.length;
      // 判断栈是否位空
      if (c != 0) {
        // 不为空计算长度
        let width = i - arr[c - 1];
        // 记录最大值
        max = Math.max(max, width);
      } else {
        // 为空就把当前索引占位 因为一个 ）是无效的
        arr.push(i);
      }
    }
  }
  return max;
};
