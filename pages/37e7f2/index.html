    全局消息组件

    笔记

    通过 Message 通过调用 通过参数 type 控制样式和 text 控制提示内容，然后经过三秒自动消失，出现和消失有动画

    # 消息提示组件样式

    文件名： XtxMessage.vue

    <template>
  <Transition name="down">
    <div class="xtx-message" :style="style[type]" v-if="show">
      <!--图标-->
      <i :class="style[type].icon" class="iconfont"></i>
      <!--提示文字-->
      <span class="text">{{ text }}</span>
    </div>
  </Transition>
</template>
<script>
import { onMounted, ref } from "vue";

export default {
  name: "XtxMessage",
  props: {
    // 消息框样式
    type: {
      type: String,
      default: "error",
    },
    // 提示内容
    text: {
      type: String,
      default: "",
    },
  },
  setup() {
    // 组件样式
    const style = {
      // 警告类型的提示
      warn: {
        icon: "icon-warning",
        color: "#E6A23C",
        backgroundColor: "rgb(253, 246, 236)",
        borderColor: "rgb(250, 236, 216)",
      },
      // 错误类型的提示
      error: {
        icon: "icon-shanchu",
        color: "#F56C6C",
        backgroundColor: "rgb(254, 240, 240)",
        borderColor: "rgb(253, 226, 226)",
      },
      // 成功类型的提示
      success: {
        icon: "icon-queren2",
        color: "#67C23A",
        backgroundColor: "rgb(240, 249, 235)",
        borderColor: "rgb(225, 243, 216)",
      },
    };
    // 默认不渲染组件
    const show = ref(false);
    onMounted(() => {
      // 组件挂载完成之后渲染 目的是做动画
      // 动画的触发必须是组件的创建和销毁
      show.value = true;
    });
    return { style, show };
  },
};
</script>

    # 通过 JS 的方式渲染 VUE 组件

    文件名：Message.js

    import XtxMessage from "./XtxMessage";
import { createVNode, render } from "vue";

// 如果通过 JS 的方式渲染 VUE 组件
// 1. 导入目标 vue 组件
// 2. 创建一个组件渲染容器
// 3. 将目标单文件vue组件渲染成虚拟 DOM 对象
// 4. 将组件的虚拟DOM对象渲染成真实 dom 对象
const container = document.createElement("div");
// 把容器渲染到body里面
document.body.appendChild(container);
// 定义定时器
let timer = null;
export default function Message({ type, text }) {
  // 将单文件组件对象转换成虚拟dom对象
  const vNode = createVNode(XtxMessage, { type, text });
  // 将虚拟DOM对象转换为真实dom对象并将其渲染到container容器中
  render(vNode, container);
  // 清除定时器
  clearTimeout(timer);
  timer = setTimeout(() => {
    // 销毁组件 组件是通过show属性判断是否渲染
    vNode.component.proxy.show = false;
    // _vnode不能自带属性，如果不设置为null 组件不会重新创建
    container._vnode = null;
  }, 3000);
}

    # 挂载到全局

    文件名：index.js

    import Message from "./Message.js";
const library = {
  install(app) {
    // 将 Message 方法添加到全局属性中
    app.config.globalProperties.$message = Message;
  },
};

export default library;

    # 使用

/**
 * type 提示框样式
 *     - success：成功
 *     - error ： 失败
 *     - warn ：警告
 * text ： 提示文字
 */
Message({ type: "success", text: "登录成功" });
