    注意

    本文档使用的是 vee-validate4.5+和 Vue3 来完成代码示例,如果拷贝代码有问题,请注意版本

    # 什么是 vee-validate?

    vee-validate 是专用于 Vue.js 的表单验证库。

    # 安装

    yarn add vee-validate@4.56  // or npm install vee-validate@4.56 --save
    1

    笔记

    因为我们使用的是 vue3 使用必须安装 vee-validate@4 + 以上的版本，因为 vee-validate@4 + 以上的版本才支持 vue3

    # 使用

    # 1. 创建表单验证对象

    // 引入 useForm
import { useForm } from "vee-validate";
// 创建表单验证对象
useFrom({
  // validationSchema 存放表单校验的集合
  validationSchema: {
    // 存放单个表单校验的方式
    // value 是输入框返回的数据
    name(value) {
      // 根据value进行表单校验
      // return '错误提示内容'
      if (!value) return "请输入用户名";
      if (!/^[a-zA-Z]\w{5,19}$/.test(value)) return "字母开头且6-20个字符";
      // return true 表示通过校验
      return true;
    },
  },
});

    # 2. 绑定输入框进行表单校验

    // 引入 useField
import { useField } from "vee-validate";
/**
 * 绑定输入框值进行校验 ---- 默认输入框输入触发表单校验
 * value ： 输入框的数据 ---- 可以通过vue提供的v-model 进行双向绑定
 * errorMessage ：表单检验不通过返回的信息
 * 'name' : 和创建表单验证对象 validationSchema 属性里面的方法名要一一对应
 */
const { value, errorMessage } = useField("name");

    # 修改校验触发的方式

    1.0 关闭校验仅点击提交时进行校验

    useField("fieldName", undefined, { validateOnValueUpdate: false });

    # 官方提供了一个优化表单校验触发的代码如下(需要时可以直接复制)

    <template>
  <div>
    <input v-on="validationListeners" :value="value" type="text" />
    <span>{{ errorMessage }}</span>
  </div>
</template>
<script>
import { computed } from "vue";
import { useField } from "vee-validate";
export default {
  setup() {
    function isRequired(value) {
      // ...
    }
    const { errorMessage, value, handleChange } = useField(
      "fieldName",
      isRequired,
      {
        validateOnValueUpdate: false,
      }
    );
    const validationListeners = computed(() => {
      // If the field is valid or have not been validated yet
      // lazy
      if (!errorMessage.value) {
        return {
          blur: handleChange,
          change: handleChange,
          // disable `shouldValidate` to avoid validating on input
          input: (e) => handleChange(e, false),
        };
      }
      // Aggressive
      return {
        blur: handleChange,
        change: handleChange,
        input: handleChange, // only switched this
      };
    });
    return {
      errorMessage,
      value,
      validationListeners,
    };
  },
};
</script>

// 初始值对象里面的键要和 validationSchema 里面的校验方法对应(校验方法绑定的那个输入框，就是设置那个输入框的默认值)

  initialValues: {
    // 设置初始值
    name: "张三",
  },

const { validate } = useField();
// validate是异步函数需要通过await 和 async处理一下
// valid ：是否通过校验
// errors: 错误提示
const { valid, errors } = await validate();
