# 简介

# 什么是 VueUse

为 Vue 2 和 Vue 3 服务的一套 Vue Composition API 的常用工具集

# 安装

yarn add @vueuse/core   // or npm i @vueuse/core

# 常见方法

# useClipboard-剪贴板

# 应用场景

点击复制等

# 基本使用

// 引入需要的方法
import { useClipboard } from "@vueuse/core";
// 定义需要复制的容器
const source = ref("Hello");
/**
 * text : 复制的文本
 * copy : 复制的方法
 * copied : 是否复制成功(false:复制成功)
 */
const { text, copy, copied } = useClipboard({ source });

# useFullscreen-全屏

# 应用场景

点击全屏按钮 让视频全屏

# 基本使用

<script setup>
import { ref } from "vue";
// 引入需要的库
import { useFullscreen } from "@vueuse/core";
// el: 全屏元素
const el = ref(null);
/**
 * toggle: 全屏切换
 * isFullscreen: 是否全屏(true:全屏)
 */
const { toggle, isFullscreen } = useFullscreen(el);
</script>

<template>
  <div class="text-center">
    <video
      ref="el"
      class="mb-2"
      src="https://vjs.zencdn.net/v/oceans.mp4"
      width="500"
      controls
    />
    <br />
    <br />
    <button @click="toggle">Go Fullscreen</button>
  </div>
</template>

# onClickOutside-点击元素以外的区域

# 应用场景

点击弹出框以外的区域关闭弹出框等

# 基本使用

<template>
  <div ref="target">Hello world</div>
</template>

<script>
import { ref } from "vue";
// 引入方法
import { onClickOutside } from "@vueuse/core";

export default {
  setup() {
    // 指定元素
    const target = ref(null);
    /**
     * target:指定元素
     * fn : 点击指定元素之外区域后的回调
     */
    onClickOutside(target, (event) => console.log(event));

    return { target };
  },
};
</script>

# useIntervalFn - 倒计时

# 应用场景

验证码倒计时

# 基本使用

import { useIntervalFn } from "@vueuse/core";
/**
 *
 * pause ：停止倒计时
 * isActive : 倒计时是否在进行
 * resume ： 重新开始倒计时
 */
const { pause, isActive, resume } = useIntervalFn(
  // 没事倒计时执行的函数
  () => {},
  // 每隔多久执行
  1000,
  // immediate 默认不执行，需要调用resume()才会开始倒计时
  { immediate: false }
);
